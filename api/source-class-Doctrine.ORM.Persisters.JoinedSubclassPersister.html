128:     {

135:         $isPostInsertId = $idGenerator->isPostInsertGenerator();

137:             ? $this->em->getClassMetadata($this->class->rootEntityName)
138:             : $this->class;

141:         $rootPersister = $this->em->getUnitOfWork()->getEntityPersister($rootClass->name);
142:         $rootTableName = $rootClass->getTableName();
143:         $rootTableStmt = $this->conn->prepare($rootPersister->getInsertSQL());
144: 
145:         // Prepare statements for sub tables.
146:         $subTableStmts = array();
147: 
148:         if ($rootClass !== $this->class) {
149:             $subTableStmts[$this->class->getTableName()] = $this->conn->prepare($this->getInsertSQL());
150:         }
151: 
152:         foreach ($this->class->parentClasses as $parentClassName) {
153:             $parentClass = $this->em->getClassMetadata($parentClassName);
154:             $parentTableName = $parentClass->getTableName();
155: 
156:             if ($parentClass !== $rootClass) {
157:                 $parentPersister = $this->em->getUnitOfWork()->getEntityPersister($parentClassName);
158:                 $subTableStmts[$parentTableName] = $this->conn->prepare($parentPersister->getInsertSQL());
159:             }
160:         }
161: 
162:         // Execute all inserts. For each entity:
163:         // 1) Insert on root table
164:         // 2) Insert on sub tables
165:         foreach ($this->queuedInserts as $entity) {
166:             $insertData = $this->prepareInsertData($entity);
167: 
168:             // Execute insert on root table
169:             $paramIndex = 1;
170: 
171:             foreach ($insertData[$rootTableName] as $columnName => $value) {
172:                 $rootTableStmt->bindValue($paramIndex++, $value, $this->columnTypes[$columnName]);
173:             }
174: 
175:             $rootTableStmt->execute();
176: 
177:             if ($isPostInsertId) {
178:                 $id = $idGenerator->generate($this->em, $entity);
179:                 $postInsertIds[$id] = $entity;
180:             } else {
181:                 $id = $this->em->getUnitOfWork()->getEntityIdentifier($entity);
182:             }
183: 
184:             // Execute inserts on subtables.
185:             // The order doesn't matter because all child tables link to the root table via FK.
186:             foreach ($subTableStmts as $tableName => $stmt) {
187:                 /** @var \Doctrine\DBAL\Statement $stmt */
188:                 $paramIndex = 1;
189:                 $data       = isset($insertData[$tableName])
190:                     ? $insertData[$tableName]
191:                     : array();
192: 
193:                 foreach ((array) $id as $idName => $idVal) {
194:                     $type = isset($this->columnTypes[$idName]) ? $this->columnTypes[$idName] : Type::STRING;
195: 
196:                     $stmt->bindValue($paramIndex++, $idVal, $type);
197:                 }
198: 
199:                 foreach ($data as $columnName => $value) {
200:                     if (!is_array($id) || !isset($id[$columnName])) {
201:                         $stmt->bindValue($paramIndex++, $value, $this->columnTypes[$columnName]);
202:                     }
203:                 }
204: 
205:                 $stmt->execute();
206:             }
207:         }
208: 
209:         $rootTableStmt->closeCursor();
210: 
211:         foreach ($subTableStmts as $stmt) {
212:             $stmt->closeCursor();
213:         }
214: 
215:         if ($this->class->isVersioned) {
216:             $this->assignDefaultVersionValue($entity, $id);
217:         }
218: 
219:         $this->queuedInserts = array();
220: 
221:         return $postInsertIds;
222:     }
