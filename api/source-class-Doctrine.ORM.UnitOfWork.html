 269:     /**
 270:      * Commits the UnitOfWork, executing all operations that have been postponed
 271:      * up to this point. The state of all managed entities will be synchronized with
 272:      * the database.
 273:      *
 274:      * The operations are executed in the following order:
 275:      *
 276:      * 1) All entity insertions
 277:      * 2) All entity updates
 278:      * 3) All collection deletions
 279:      * 4) All collection updates
 280:      * 5) All entity deletions
 281:      *
 282:      * @param null|object|array $entity
 283:      *
 284:      * @return void
 285:      *
 286:      * @throws \Exception
 287:      */
 288:     public function commit($entity = null)
 289:     {
 290:         // Raise preFlush
 291:         if ($this->evm->hasListeners(Events::preFlush)) {
 292:             $this->evm->dispatchEvent(Events::preFlush, new PreFlushEventArgs($this->em));
 293:         }
 294: 
 295:         // Compute changes done since last commit.
 296:         if ($entity === null) {
 297:             $this->computeChangeSets();
 298:         } elseif (is_object($entity)) {
 299:             $this->computeSingleEntityChangeSet($entity);
 300:         } elseif (is_array($entity)) {
 301:             foreach ($entity as $object) {
 302:                 $this->computeSingleEntityChangeSet($object);
 303:             }
 304:         }
 305: 
 306:         if ( ! ($this->entityInsertions ||
 307:                 $this->entityDeletions ||
 308:                 $this->entityUpdates ||
 309:                 $this->collectionUpdates ||
 310:                 $this->collectionDeletions ||
 311:                 $this->orphanRemovals)) {
 312:             $this->dispatchOnFlushEvent();
 313:             $this->dispatchPostFlushEvent();
 314: 
 315:             return; // Nothing to do.
 316:         }
 317: 
 318:         if ($this->orphanRemovals) {
 319:             foreach ($this->orphanRemovals as $orphan) {
 320:                 $this->remove($orphan);
 321:             }
 322:         }
 323: 
 324:         $this->dispatchOnFlushEvent();
 325: 
 326:         // Now we need a commit order to maintain referential integrity
 327:         $commitOrder = $this->getCommitOrder();
 328: 
 329:         $conn = $this->em->getConnection();
 330:         $conn->beginTransaction();
 331: 
 332:         try {
 333:             if ($this->entityInsertions) {
 334:                 foreach ($commitOrder as $class) {
 335:                     $this->executeInserts($class);
 336:                 }
 337:             }
 338: 
 339:             if ($this->entityUpdates) {
 340:                 foreach ($commitOrder as $class) {
 341:                     $this->executeUpdates($class);
 342:                 }
 343:             }
 344: 
 345:             // Extra updates that were requested by persisters.
 346:             if ($this->extraUpdates) {
 347:                 $this->executeExtraUpdates();
 348:             }
 349: 
 350:             // Collection deletions (deletions of complete collections)
 351:             foreach ($this->collectionDeletions as $collectionToDelete) {
 352:                 $this->getCollectionPersister($collectionToDelete->getMapping())->delete($collectionToDelete);
 353:             }
 354:             // Collection updates (deleteRows, updateRows, insertRows)
 355:             foreach ($this->collectionUpdates as $collectionToUpdate) {
 356:                 $this->getCollectionPersister($collectionToUpdate->getMapping())->update($collectionToUpdate);
 357:             }
 358: 
 359:             // Entity deletions come last and need to be in reverse commit order
 360:             if ($this->entityDeletions) {
 361:                 for ($count = count($commitOrder), $i = $count - 1; $i >= 0; --$i) {
 362:                     $this->executeDeletions($commitOrder[$i]);
 363:                 }
 364:             }
 365: 
 366:             $conn->commit();
 367:         } catch (Exception $e) {
 368:             $this->em->close();
 369:             $conn->rollback();
 370: 
 371:             throw $e;
 372:         }
 373: 
 374:         // Take new snapshots from visited collections
 375:         foreach ($this->visitedCollections as $coll) {
 376:             $coll->takeSnapshot();
 377:         }
 378: 
 379:         $this->dispatchPostFlushEvent();
 380: 
 381:         // Clear up
 382:         $this->entityInsertions =
 383:         $this->entityUpdates =
 384:         $this->entityDeletions =
 385:         $this->extraUpdates =
 386:         $this->entityChangeSets =
 387:         $this->collectionUpdates =
 388:         $this->collectionDeletions =
 389:         $this->visitedCollections =
 390:         $this->scheduledForDirtyCheck =
 391:         $this->orphanRemovals = array();
 392:     }

 408:     /**
 409:      * Only flushes the given entity according to a ruleset that keeps the UoW consistent.
 410:      *
 411:      * 1. All entities scheduled for insertion, (orphan) removals and changes in collections are processed as well!
 412:      * 2. Read Only entities are skipped.
 413:      * 3. Proxies are skipped.
 414:      * 4. Only if entity is properly managed.
 415:      *
 416:      * @param object $entity
 417:      *
 418:      * @return void
 419:      *
 420:      * @throws \InvalidArgumentException
 421:      */
 422:     private function computeSingleEntityChangeSet($entity)
 423:     {
 424:         $state = $this->getEntityState($entity);
 425: 
 426:         if ($state !== self::STATE_MANAGED && $state !== self::STATE_REMOVED) {
 427:             throw new \InvalidArgumentException("Entity has to be managed or scheduled for removal for single computation " . self::objToStr($entity));
 428:         }
 429: 
 430:         $class = $this->em->getClassMetadata(get_class($entity));
 431: 
 432:         if ($state === self::STATE_MANAGED && $class->isChangeTrackingDeferredImplicit()) {
 433:             $this->persist($entity);
 434:         }
 435: 
 436:         // Compute changes for INSERTed entities first. This must always happen even in this case.
 437:         $this->computeScheduleInsertsChangeSets();
 438: 
 439:         if ($class->isReadOnly) {
 440:             return;
 441:         }
 442: 
 443:         // Ignore uninitialized proxy objects
 444:         if ($entity instanceof Proxy && ! $entity->__isInitialized__) {
 445:             return;
 446:         }
 447: 
 448:         // Only MANAGED entities that are NOT SCHEDULED FOR INSERTION are processed here.
 449:         $oid = spl_object_hash($entity);
 450: 
 451:         if ( ! isset($this->entityInsertions[$oid]) && isset($this->entityStates[$oid])) {
 452:             $this->computeChangeSet($class, $entity);
 453:         }
 454:     }

 487:     /**
 488:      * Computes the changes that happened to a single entity.
 489:      *
 490:      * Modifies/populates the following properties:
 491:      *
 492:      * {@link _originalEntityData}
 493:      * If the entity is NEW or MANAGED but not yet fully persisted (only has an id)
 494:      * then it was not fetched from the database and therefore we have no original
 495:      * entity data yet. All of the current entity data is stored as the original entity data.
 496:      *
 497:      * {@link _entityChangeSets}
 498:      * The changes detected on all properties of the entity are stored there.
 499:      * A change is a tuple array where the first entry is the old value and the second
 500:      * entry is the new value of the property. Changesets are used by persisters
 501:      * to INSERT/UPDATE the persistent entity state.
 502:      *
 503:      * {@link _entityUpdates}
 504:      * If the entity is already fully MANAGED (has been fetched from the database before)
 505:      * and any changes to its properties are detected, then a reference to the entity is stored
 506:      * there to mark it for an update.
 507:      *
 508:      * {@link _collectionDeletions}
 509:      * If a PersistentCollection has been de-referenced in a fully MANAGED entity,
 510:      * then this collection is marked for deletion.
 511:      *
 512:      * @ignore
 513:      *
 514:      * @internal Don't call from the outside.
 515:      *
 516:      * @param ClassMetadata $class The class descriptor of the entity.
 517:      * @param object $entity The entity for which to compute the changes.
 518:      *
 519:      * @return void
 520:      */
 521:     public function computeChangeSet(ClassMetadata $class, $entity)
 522:     {
 523:         $oid = spl_object_hash($entity);
 524: 
 525:         if (isset($this->readOnlyObjects[$oid])) {
 526:             return;
 527:         }
 528: 
 529:         if ( ! $class->isInheritanceTypeNone()) {
 530:             $class = $this->em->getClassMetadata(get_class($entity));
 531:         }
 532: 
 533:         $invoke = $this->listenersInvoker->getSubscribedSystems($class, Events::preFlush) & ~ListenersInvoker::INVOKE_MANAGER;
 534: 
 535:         if ($invoke !== ListenersInvoker::INVOKE_NONE) {
 536:             $this->listenersInvoker->invoke($class, Events::preFlush, $entity, new PreFlushEventArgs($this->em), $invoke);
 537:         }
 538: 
 539:         $actualData = array();
 540: 
 541:         foreach ($class->reflFields as $name => $refProp) {
 542:             $value = $refProp->getValue($entity);
 543: 
 544:             if ($class->isCollectionValuedAssociation($name) && $value !== null) {
 545:                 if ($value instanceof PersistentCollection) {
 546:                     if ($value->getOwner() === $entity) {
 547:                         continue;
 548:                     }
 549: 
 550:                     $value = new ArrayCollection($value->getValues());
 551:                 }
 552: 
 553:                 // If $value is not a Collection then use an ArrayCollection.
 554:                 if ( ! $value instanceof Collection) {
 555:                     $value = new ArrayCollection($value);
 556:                 }
 557: 
 558:                 $assoc = $class->associationMappings[$name];
 559: 
 560:                 // Inject PersistentCollection
 561:                 $value = new PersistentCollection(
 562:                     $this->em, $this->em->getClassMetadata($assoc['targetEntity']), $value
 563:                 );
 564:                 $value->setOwner($entity, $assoc);
 565:                 $value->setDirty( ! $value->isEmpty());
 566: 
 567:                 $class->reflFields[$name]->setValue($entity, $value);
 568: 
 569:                 $actualData[$name] = $value;
 570: 
 571:                 continue;
 572:             }
 573: 
 574:             if (( ! $class->isIdentifier($name) || ! $class->isIdGeneratorIdentity()) && ($name !== $class->versionField)) {
 575:                 $actualData[$name] = $value;
 576:             }
 577:         }
 578: 
 579:         if ( ! isset($this->originalEntityData[$oid])) {
 580:             // Entity is either NEW or MANAGED but not yet fully persisted (only has an id).
 581:             // These result in an INSERT.
 582:             $this->originalEntityData[$oid] = $actualData;
 583:             $changeSet = array();
 584: 
 585:             foreach ($actualData as $propName => $actualValue) {
 586:                 if ( ! isset($class->associationMappings[$propName])) {
 587:                     $changeSet[$propName] = array(null, $actualValue);
 588: 
 589:                     continue;
 590:                 }
 591: 
 592:                 $assoc = $class->associationMappings[$propName];
 593: 
 594:                 if ($assoc['isOwningSide'] && $assoc['type'] & ClassMetadata::TO_ONE) {
 595:                     $changeSet[$propName] = array(null, $actualValue);
 596:                 }
 597:             }
 598: 
 599:             $this->entityChangeSets[$oid] = $changeSet;
 600:         } else {
 601:             // Entity is "fully" MANAGED: it was already fully persisted before
 602:             // and we have a copy of the original data
 603:             $originalData           = $this->originalEntityData[$oid];
 604:             $isChangeTrackingNotify = $class->isChangeTrackingNotify();
 605:             $changeSet              = ($isChangeTrackingNotify && isset($this->entityChangeSets[$oid]))
 606:                 ? $this->entityChangeSets[$oid]
 607:                 : array();
 608: 
 609:             foreach ($actualData as $propName => $actualValue) {
 610:                 // skip field, its a partially omitted one!
 611:                 if ( ! (isset($originalData[$propName]) || array_key_exists($propName, $originalData))) {
 612:                     continue;
 613:                 }
 614: 
 615:                 $orgValue = $originalData[$propName];
 616: 
 617:                 // skip if value haven't changed
 618:                 if ($orgValue === $actualValue) {
 619:                     continue;
 620:                 }
 621: 
 622:                 // if regular field
 623:                 if ( ! isset($class->associationMappings[$propName])) {
 624:                     if ($isChangeTrackingNotify) {
 625:                         continue;
 626:                     }
 627: 
 628:                     $changeSet[$propName] = array($orgValue, $actualValue);
 629: 
 630:                     continue;
 631:                 }
 632: 
 633:                 $assoc = $class->associationMappings[$propName];
 634: 
 635:                 // Persistent collection was exchanged with the "originally"
 636:                 // created one. This can only mean it was cloned and replaced
 637:                 // on another entity.
 638:                 if ($actualValue instanceof PersistentCollection) {
 639:                     $owner = $actualValue->getOwner();
 640:                     if ($owner === null) { // cloned
 641:                         $actualValue->setOwner($entity, $assoc);
 642:                     } else if ($owner !== $entity) { // no clone, we have to fix
 643:                         if (!$actualValue->isInitialized()) {
 644:                             $actualValue->initialize(); // we have to do this otherwise the cols share state
 645:                         }
 646:                         $newValue = clone $actualValue;
 647:                         $newValue->setOwner($entity, $assoc);
 648:                         $class->reflFields[$propName]->setValue($entity, $newValue);
 649:                     }
 650:                 }
 651: 
 652:                 if ($orgValue instanceof PersistentCollection) {
 653:                     // A PersistentCollection was de-referenced, so delete it.
 654:                     $coid = spl_object_hash($orgValue);
 655: 
 656:                     if (isset($this->collectionDeletions[$coid])) {
 657:                         continue;
 658:                     }
 659: 
 660:                     $this->collectionDeletions[$coid] = $orgValue;
 661:                     $changeSet[$propName] = $orgValue; // Signal changeset, to-many assocs will be ignored.
 662: 
 663:                     continue;
 664:                 }
 665: 
 666:                 if ($assoc['type'] & ClassMetadata::TO_ONE) {
 667:                     if ($assoc['isOwningSide']) {
 668:                         $changeSet[$propName] = array($orgValue, $actualValue);
 669:                     }
 670: 
 671:                     if ($orgValue !== null && $assoc['orphanRemoval']) {
 672:                         $this->scheduleOrphanRemoval($orgValue);
 673:                     }
 674:                 }
 675:             }
 676: 
 677:             if ($changeSet) {
 678:                 $this->entityChangeSets[$oid]   = $changeSet;
 679:                 $this->originalEntityData[$oid] = $actualData;
 680:                 $this->entityUpdates[$oid]      = $entity;
 681:             }
 682:         }
 683: 
 684:         // Look for changes in associations of the entity
 685:         foreach ($class->associationMappings as $field => $assoc) {
 686:             if (($val = $class->reflFields[$field]->getValue($entity)) !== null) {
 687:                 $this->computeAssociationChanges($assoc, $val);
 688:                 if (!isset($this->entityChangeSets[$oid]) &&
 689:                     $assoc['isOwningSide'] &&
 690:                     $assoc['type'] == ClassMetadata::MANY_TO_MANY &&
 691:                     $val instanceof PersistentCollection &&
 692:                     $val->isDirty()) {
 693:                     $this->entityChangeSets[$oid]   = array();
 694:                     $this->originalEntityData[$oid] = $actualData;
 695:                     $this->entityUpdates[$oid]      = $entity;
 696:                 }
 697:             }
 698:         }
 699:     }

 701:     /**
 702:      * Computes all the changes that have been done to entities and collections
 703:      * since the last commit and stores these changes in the _entityChangeSet map
 704:      * temporarily for access by the persisters, until the UoW commit is finished.
 705:      *
 706:      * @return void
 707:      */
 708:     public function computeChangeSets()
 709:     {
 710:         // Compute changes for INSERTed entities first. This must always happen.
 711:         $this->computeScheduleInsertsChangeSets();
 712: 
 713:         // Compute changes for other MANAGED entities. Change tracking policies take effect here.
 714:         foreach ($this->identityMap as $className => $entities) {
 715:             $class = $this->em->getClassMetadata($className);
 716: 
 717:             // Skip class if instances are read-only
 718:             if ($class->isReadOnly) {
 719:                 continue;
 720:             }
 721: 
 722:             // If change tracking is explicit or happens through notification, then only compute
 723:             // changes on entities of that type that are explicitly marked for synchronization.
 724:             switch (true) {
 725:                 case ($class->isChangeTrackingDeferredImplicit()):
 726:                     $entitiesToProcess = $entities;
 727:                     break;
 728: 
 729:                 case (isset($this->scheduledForDirtyCheck[$className])):
 730:                     $entitiesToProcess = $this->scheduledForDirtyCheck[$className];
 731:                     break;
 732: 
 733:                 default:
 734:                     $entitiesToProcess = array();
 735: 
 736:             }
 737: 
 738:             foreach ($entitiesToProcess as $entity) {
 739:                 // Ignore uninitialized proxy objects
 740:                 if ($entity instanceof Proxy && ! $entity->__isInitialized__) {
 741:                     continue;
 742:                 }
 743: 
 744:                 // Only MANAGED entities that are NOT SCHEDULED FOR INSERTION are processed here.
 745:                 $oid = spl_object_hash($entity);
 746: 
 747:                 if ( ! isset($this->entityInsertions[$oid]) && isset($this->entityStates[$oid])) {
 748:                     $this->computeChangeSet($class, $entity);
 749:                 }
 750:             }
 751:         }
 752:     }

 754:     /**
 755:      * Computes the changes of an association.
 756:      *
 757:      * @param array $assoc
 758:      * @param mixed $value The value of the association.
 759:      *
 760:      * @throws ORMInvalidArgumentException
 761:      * @throws ORMException
 762:      *
 763:      * @return void
 764:      */
 765:     private function computeAssociationChanges($assoc, $value)
 766:     {
 767:         if ($value instanceof Proxy && ! $value->__isInitialized__) {
 768:             return;
 769:         }
 770: 
 771:         if ($value instanceof PersistentCollection && $value->isDirty()) {
 772:             $coid = spl_object_hash($value);
 773: 
 774:             if ($assoc['isOwningSide']) {
 775:                 $this->collectionUpdates[$coid] = $value;
 776:             }
 777: 
 778:             $this->visitedCollections[$coid] = $value;
 779:         }
 780: 
 781:         // Look through the entities, and in any of their associations,
 782:         // for transient (new) entities, recursively. ("Persistence by reachability")
 783:         // Unwrap. Uninitialized collections will simply be empty.
 784:         $unwrappedValue = ($assoc['type'] & ClassMetadata::TO_ONE) ? array($value) : $value->unwrap();
 785:         $targetClass    = $this->em->getClassMetadata($assoc['targetEntity']);
 786: 
 787:         foreach ($unwrappedValue as $key => $entry) {
 788:             $state = $this->getEntityState($entry, self::STATE_NEW);
 789: 
 790:             if ( ! ($entry instanceof $assoc['targetEntity'])) {
 791:                 throw new ORMException(
 792:                     sprintf(
 793:                         'Found entity of type %s on association %s#%s, but expecting %s',
 794:                         get_class($entry),
 795:                         $assoc['sourceEntity'],
 796:                         $assoc['fieldName'],
 797:                         $targetClass->name
 798:                     )
 799:                 );
 800:             }
 801: 
 802:             switch ($state) {
 803:                 case self::STATE_NEW:
 804:                     if ( ! $assoc['isCascadePersist']) {
 805:                         throw ORMInvalidArgumentException::newEntityFoundThroughRelationship($assoc, $entry);
 806:                     }
 807: 
 808:                     $this->persistNew($targetClass, $entry);
 809:                     $this->computeChangeSet($targetClass, $entry);
 810:                     break;
 811: 
 812:                 case self::STATE_REMOVED:
 813:                     // Consume the $value as array (it's either an array or an ArrayAccess)
 814:                     // and remove the element from Collection.
 815:                     if ($assoc['type'] & ClassMetadata::TO_MANY) {
 816:                         unset($value[$key]);
 817:                     }
 818:                     break;
 819: 
 820:                 case self::STATE_DETACHED:
 821:                     // Can actually not happen right now as we assume STATE_NEW,
 822:                     // so the exception will be raised from the DBAL layer (constraint violation).
 823:                     throw ORMInvalidArgumentException::detachedEntityFoundThroughRelationship($assoc, $entry);
 824:                     break;
 825: 
 826:                 default:
 827:                     // MANAGED associated entities are already taken into account
 828:                     // during changeset calculation anyway, since they are in the identity map.
 829:             }
 830:         }
 831:     }

 836:      *
 837:      * @return void

 843:

 845:             $this->listenersInvoker->invoke($class, Events::prePersist, $entity, new LifecycleEventArgs($entity, $this->em), $invoke);

 848:         $idGen = $class->idGenerator;

 867:     /**
 868:      * INTERNAL:
 869:      * Computes the changeset of an individual entity, independently of the
 870:      * computeChangeSets() routine that is used at the beginning of a UnitOfWork#commit().
 871:      *
 872:      * The passed entity must be a managed entity. If the entity already has a change set
 873:      * because this method is invoked during a commit cycle then the change sets are added.
 874:      * whereby changes detected in this method prevail.
 875:      *
 876:      * @ignore
 877:      *
 878:      * @param ClassMetadata $class  The class descriptor of the entity.
 879:      * @param object        $entity The entity for which to (re)calculate the change set.
 880:      *
 881:      * @return void
 882:      *
 883:      * @throws ORMInvalidArgumentException If the passed entity is not MANAGED.
 884:      */
 885:     public function recomputeSingleEntityChangeSet(ClassMetadata $class, $entity)
 886:     {
 887:         $oid = spl_object_hash($entity);
 888: 
 889:         if ( ! isset($this->entityStates[$oid]) || $this->entityStates[$oid] != self::STATE_MANAGED) {
 890:             throw ORMInvalidArgumentException::entityNotManaged($entity);
 891:         }
 892: 
 893:         // skip if change tracking is "NOTIFY"
 894:         if ($class->isChangeTrackingNotify()) {
 895:             return;
 896:         }
 897: 
 898:         if ( ! $class->isInheritanceTypeNone()) {
 899:             $class = $this->em->getClassMetadata(get_class($entity));
 900:         }
 901: 
 902:         $actualData = array();
 903: 
 904:         foreach ($class->reflFields as $name => $refProp) {
 905:             if (( ! $class->isIdentifier($name) || ! $class->isIdGeneratorIdentity())
 906:                 && ($name !== $class->versionField)
 907:                 && ! $class->isCollectionValuedAssociation($name)) {
 908:                 $actualData[$name] = $refProp->getValue($entity);
 909:             }
 910:         }
 911: 
 912:         if ( ! isset($this->originalEntityData[$oid])) {
 913:             throw new \RuntimeException('Cannot call recomputeSingleEntityChangeSet before computeChangeSet on an entity.');
 914:         }
 915: 
 916:         $originalData = $this->originalEntityData[$oid];
 917:         $changeSet = array();
 918: 
 919:         foreach ($actualData as $propName => $actualValue) {
 920:             $orgValue = isset($originalData[$propName]) ? $originalData[$propName] : null;
 921: 
 922:             if ($orgValue !== $actualValue) {
 923:                 $changeSet[$propName] = array($orgValue, $actualValue);
 924:             }
 925:         }
 926: 
 927:         if ($changeSet) {
 928:             $this->entityChangeSets[$oid] = (isset($this->entityChangeSets[$oid]))
 929:                 ? array_merge($this->entityChangeSets[$oid], $changeSet)
 930:                 : $changeSet;
 931: 
 932:             $this->originalEntityData[$oid] = $actualData;
 933:             $this->entityUpdates[$oid]      = $entity;
 934:         }
 935:     }

 937:     /**
 938:      * Executes all entity insertions for entities of the specified type.
 939:      *
 940:      * @param \Doctrine\ORM\Mapping\ClassMetadata $class
 941:      *
 942:      * @return void
 943:      */
 944:     private function executeInserts($class)
 945:     {
 946:         $entities   = array();
 947:         $className  = $class->name;
 948:         $persister  = $this->getEntityPersister($className);
 949:         $invoke     = $this->listenersInvoker->getSubscribedSystems($class, Events::postPersist);
 950: 
 951:         foreach ($this->entityInsertions as $oid => $entity) {
 952:             if ($this->em->getClassMetadata(get_class($entity))->name !== $className) {
 953:                 continue;
 954:             }
 955: 
 956:             $persister->addInsert($entity);
 957: 
 958:             unset($this->entityInsertions[$oid]);
 959: 
 960:             if ($invoke !== ListenersInvoker::INVOKE_NONE) {
 961:                 $entities[] = $entity;
 962:             }
 963:         }
 964: 
 965:         $postInsertIds = $persister->executeInserts();
 966: 
 967:         if ($postInsertIds) {
 968:             // Persister returned post-insert IDs
 969:             foreach ($postInsertIds as $id => $entity) {
 970:                 $oid     = spl_object_hash($entity);
 971:                 $idField = $class->identifier[0];
 972: 
 973:                 $class->reflFields[$idField]->setValue($entity, $id);
 974: 
 975:                 $this->entityIdentifiers[$oid] = array($idField => $id);
 976:                 $this->entityStates[$oid] = self::STATE_MANAGED;
 977:                 $this->originalEntityData[$oid][$idField] = $id;
 978: 
 979:                 $this->addToIdentityMap($entity);
 980:             }
 981:         }
 982: 
 983:         foreach ($entities as $entity) {
 984:             $this->listenersInvoker->invoke($class, Events::postPersist, $entity, new LifecycleEventArgs($entity, $this->em), $invoke);
 985:         }
 986:     }

 988:     /**
 989:      * Executes all entity updates for entities of the specified type.
 990:      *
 991:      * @param \Doctrine\ORM\Mapping\ClassMetadata $class
 992:      *
 993:      * @return void
 994:      */
 995:     private function executeUpdates($class)
 996:     {
 997:         $className          = $class->name;
 998:         $persister          = $this->getEntityPersister($className);
 999:         $preUpdateInvoke    = $this->listenersInvoker->getSubscribedSystems($class, Events::preUpdate);
1000:         $postUpdateInvoke   = $this->listenersInvoker->getSubscribedSystems($class, Events::postUpdate);
1001: 
1002:         foreach ($this->entityUpdates as $oid => $entity) {
1003:             if ($this->em->getClassMetadata(get_class($entity))->name !== $className) {
1004:                 continue;
1005:             }
1006: 
1007:             if ($preUpdateInvoke != ListenersInvoker::INVOKE_NONE) {
1008:                 $this->listenersInvoker->invoke($class, Events::preUpdate, $entity, new PreUpdateEventArgs($entity, $this->em, $this->entityChangeSets[$oid]), $preUpdateInvoke);
1009:                 $this->recomputeSingleEntityChangeSet($class, $entity);
1010:             }
1011: 
1012:             if ( ! empty($this->entityChangeSets[$oid])) {
1013:                 $persister->update($entity);
1014:             }
1015: 
1016:             unset($this->entityUpdates[$oid]);
1017: 
1018:             if ($postUpdateInvoke != ListenersInvoker::INVOKE_NONE) {
1019:                 $this->listenersInvoker->invoke($class, Events::postUpdate, $entity, new LifecycleEventArgs($entity, $this->em), $postUpdateInvoke);
1020:             }
1021:         }
1022:     }

1028:      *
1029:      * @return void
1030:      */
1031:     private function executeDeletions($class)
1032:     {

1036: 
1037:         foreach ($this->entityDeletions as $oid => $entity) {

1039:                 continue;
1040:             }

1053:             //$this->entityStates[$oid] = self::STATE_NEW;
1054:             if ( ! $class->isIdentifierNatural()) {
1055:                 $class->reflFields[$class->identifier[0]]->setValue($entity, null);
1056:             }
1057: 
1058:             if ($invoke !== ListenersInvoker::INVOKE_NONE) {

1064:     /**
1065:      * Gets the commit order.
1066:      *
1067:      * @param array|null $entityChangeSet
1068:      *
1069:      * @return array
1070:      */
1071:     private function getCommitOrder(array $entityChangeSet = null)
1072:     {
1073:         if ($entityChangeSet === null) {
1074:             $entityChangeSet = array_merge($this->entityInsertions, $this->entityUpdates, $this->entityDeletions);
1075:         }
1076: 
1077:         $calc = $this->getCommitOrderCalculator();
1078: 
1079:         // See if there are any new classes in the changeset, that are not in the
1080:         // commit order graph yet (don't have a node).
1081:         // We have to inspect changeSet to be able to correctly build dependencies.
1082:         // It is not possible to use IdentityMap here because post inserted ids
1083:         // are not yet available.
1084:         $newNodes = array();
1085: 
1086:         foreach ($entityChangeSet as $entity) {
1087:             $className = $this->em->getClassMetadata(get_class($entity))->name;
1088: 
1089:             if ($calc->hasClass($className)) {
1090:                 continue;
1091:             }
1092: 
1093:             $class = $this->em->getClassMetadata($className);
1094:             $calc->addClass($class);
1095: 
1096:             $newNodes[] = $class;
1097:         }
1098: 
1099:         // Calculate dependencies for new nodes
1100:         while ($class = array_pop($newNodes)) {
1101:             foreach ($class->associationMappings as $assoc) {
1102:                 if ( ! ($assoc['isOwningSide'] && $assoc['type'] & ClassMetadata::TO_ONE)) {
1103:                     continue;
1104:                 }
1105: 
1106:                 $targetClass = $this->em->getClassMetadata($assoc['targetEntity']);
1107: 
1108:                 if ( ! $calc->hasClass($targetClass->name)) {
1109:                     $calc->addClass($targetClass);
1110: 
1111:                     $newNodes[] = $targetClass;
1112:                 }
1113: 
1114:                 $calc->addDependency($targetClass, $class);
1115: 
1116:                 // If the target class has mapped subclasses, these share the same dependency.
1117:                 if ( ! $targetClass->subClasses) {
1118:                     continue;
1119:                 }
1120: 
1121:                 foreach ($targetClass->subClasses as $subClassName) {
1122:                     $targetSubClass = $this->em->getClassMetadata($subClassName);
1123: 
1124:                     if ( ! $calc->hasClass($subClassName)) {
1125:                         $calc->addClass($targetSubClass);
1126: 
1127:                         $newNodes[] = $targetSubClass;
1128:                     }
1129: 
1130:                     $calc->addDependency($targetSubClass, $class);
1131:                 }
1132:             }
1133:         }
1134: 
1135:         return $calc->getCommitOrder();
1136:     }

1760:     /**

1769:      *

1771:      *         attribute and the version check against the managed copy fails.
1772:      * @throws ORMInvalidArgumentException If the entity instance is NEW.
1773:      * @throws EntityNotFoundException
1774:      */
1775:     private function doMerge($entity, array &$visited, $prevManagedCopy = null, $assoc = null)
1776:     {
1777:         $oid = spl_object_hash($entity);
1778: 
1779:         if (isset($visited[$oid])) {
1780:             return $visited[$oid]; // Prevent infinite recursion
1781:         }
1782: 
1783:         $visited[$oid] = $entity; // mark visited
1784: 
1785:         $class = $this->em->getClassMetadata(get_class($entity));
1786: 
1787:         // First we assume DETACHED, although it can still be NEW but we can avoid
1788:         // an extra db-roundtrip this way. If it is not MANAGED but has an identity,
1789:         // we need to fetch it from the db anyway in order to merge.
1790:         // MANAGED entities are ignored by the merge operation.
1791:         $managedCopy = $entity;
1792: 
1793:         if ($this->getEntityState($entity, self::STATE_DETACHED) !== self::STATE_MANAGED) {
1794:             if ($entity instanceof Proxy && ! $entity->__isInitialized()) {
1795:                 $this->em->getProxyFactory()->resetUninitializedProxy($entity);
1796:                 $entity->__load();
1797:             }
1798: 
1799:             // Try to look the entity up in the identity map.

1802:             // If there is no ID, it is actually NEW.
1803:             if ( ! $id) {

1805: 
1806:                 $this->persistNew($class, $managedCopy);
1807:             } else {

1809:                     ? $this->flattenIdentifier($class, $id)
1810:                     : $id;

1812:                 $managedCopy = $this->tryGetById($flatId, $class->rootEntityName);
1813: 
1814:                 if ($managedCopy) {
1815:                     // We have the entity in-memory already, just make sure its not removed.
1816:                     if ($this->getEntityState($managedCopy) == self::STATE_REMOVED) {
1817:                         throw ORMInvalidArgumentException::entityIsRemoved($managedCopy, "merge");
1818:                     }
1819:                 } else {
1820:                     // We need to fetch the managed copy in order to merge.
1821:                     $managedCopy = $this->em->find($class->name, $flatId);
1822:                 }
1823: 
1824:                 if ($managedCopy === null) {
1825:                     // If the identifier is ASSIGNED, it is NEW, otherwise an error
1826:                     // since the managed entity was not found.
1827:                     if ( ! $class->isIdentifierNatural()) {
1828:                         throw new EntityNotFoundException;
1829:                     }

1831:                     $managedCopy = $this->newInstance($class);
1832:                     $class->setIdentifierValues($managedCopy, $id);
1833: 
1834:                     $this->persistNew($class, $managedCopy);
1835:                 } else {
1836:                     if ($managedCopy instanceof Proxy && ! $managedCopy->__isInitialized__) {
1837:                         $managedCopy->__load();
1838:                     }
1839:                 }
1840:             }

1847:                 // Throw exception if versions don't match.

1849:                     throw OptimisticLockException::lockFailedVersionMismatch($entity, $entityVersion, $managedCopyVersion);
1850:                 }
1851:             }
1852:

1854:             foreach ($class->reflClass->getProperties() as $prop) {

1856:                 $prop->setAccessible(true);
1857:                 if ( ! isset($class->associationMappings[$name])) {
1858:                     if ( ! $class->isIdentifier($name)) {
1859:                         $prop->setValue($managedCopy, $prop->getValue($entity));
1860:                     }
1861:                 } else {
1862:                     $assoc2 = $class->associationMappings[$name];
1863:                     if ($assoc2['type'] & ClassMetadata::TO_ONE) {
1864:                         $other = $prop->getValue($entity);
1865:                         if ($other === null) {
1866:                             $prop->setValue($managedCopy, null);
1867:                         } else if ($other instanceof Proxy && !$other->__isInitialized__) {
1868:                             // do not merge fields marked lazy that have not been fetched.
1869:                             continue;
1870:                         } else if ( ! $assoc2['isCascadeMerge']) {
1871:                             if ($this->getEntityState($other) === self::STATE_DETACHED) {
1872:                                 $targetClass = $this->em->getClassMetadata($assoc2['targetEntity']);
1873:                                 $relatedId = $targetClass->getIdentifierValues($other);
1874: 
1875:                                 if ($targetClass->subClasses) {
1876:                                     $other = $this->em->find($targetClass->name, $relatedId);
1877:                                 } else {
1878:                                     $other = $this->em->getProxyFactory()->getProxy($assoc2['targetEntity'], $relatedId);
1879:                                     $this->registerManaged($other, $relatedId, array());
1880:                                 }
1881:                             }
1882: 
1883:                             $prop->setValue($managedCopy, $other);
1884:                         }
1885:                     } else {
1886:                         $mergeCol = $prop->getValue($entity);
1887:                         if ($mergeCol instanceof PersistentCollection && !$mergeCol->isInitialized()) {
1888:                             // do not merge fields marked lazy that have not been fetched.
1889:                             // keep the lazy persistent collection of the managed copy.
1890:                             continue;
1891:                         }
1892: 
1893:                         $managedCol = $prop->getValue($managedCopy);
1894:                         if (!$managedCol) {
1895:                             $managedCol = new PersistentCollection($this->em,
1896:                                     $this->em->getClassMetadata($assoc2['targetEntity']),
1897:                                     new ArrayCollection
1898:                                     );
1899:                             $managedCol->setOwner($managedCopy, $assoc2);
1900:                             $prop->setValue($managedCopy, $managedCol);
1901:                             $this->originalEntityData[$oid][$name] = $managedCol;
1902:                         }
1903:                         if ($assoc2['isCascadeMerge']) {
1904:                             $managedCol->initialize();
1905:

1907:                             if (!$managedCol->isEmpty() && $managedCol !== $mergeCol) {
